```java
package com.auction.service.ai;

import com.auction.config.DoubaoConfig;
import com.fasterxml.jackson.databind.ObjectMapper;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpStatus;
import org.springframework.http.MediaType;
import org.springframework.stereotype.Component;
import org.springframework.web.reactive.function.client.WebClient;
import reactor.core.publisher.Mono;
import java.time.Duration;
import java.util.HashMap;
import java.util.Map;

@Slf4j
@Component
@RequiredArgsConstructor
public class DoubaoClient {

    private final DoubaoConfig config;
    private final ObjectMapper objectMapper = new ObjectMapper();

    private WebClient webClient;

    private WebClient getWebClient() {
        if (webClient == null) {
            webClient = WebClient.builder()
                .baseUrl(config.getEndpoint())
                .defaultHeader(HttpHeaders.AUTHORIZATION, "Bearer " + config.getKey())
                .defaultHeader(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_JSON_VALUE)
                .build();
        }
        return webClient;
    }

    /**
     * 调用豆包 API
     */
    public String chat(String userMessage) {
        DoubaoRequest request = DoubaoRequest.create(userMessage);

        try {
            String response = getWebClient().post()
                .bodyValue(request)
                .retrieve()
                .onStatus(HttpStatus::isError, clientResponse -> {
                    log.error("豆包 API 调用失败: status={}", clientResponse.statusCode());
                    return clientResponse.bodyToMono(String.class)
                        .flatMap(body -> Mono.error(new RuntimeException("豆包 API 调用失败: " + body)));
                })
                .bodyToMono(String.class)
                .timeout(Duration.ofMillis(config.getTimeout()))
                .block();

            DoubaoResponse doubaoResponse = objectMapper.readValue(response, DoubaoResponse.class);

            log.info("豆包 API 调用成功: tokens={}",
                doubaoResponse.getUsage() != null ? doubaoResponse.getUsage().getTotalTokens() : "N/A");

            return doubaoResponse.getContent();

        } catch (Exception e) {
            log.error("豆包 API 调用异常", e);
            throw new RuntimeException("AI 服务暂时不可用，请稍后再试", e);
        }
    }

    /**
     * 调用豆包 API 并解析 JSON 响应
     */
    public <T> T chatJson(String userMessage, Class<T> responseType) {
        String content = chat(userMessage);

        try {
            // 提取 JSON 内容（去除可能的 markdown 代码块标记）
            String jsonContent = content;
            if (content.contains("```json")) {
                jsonContent = content.substring(
                    content.indexOf("```json") + 7,
                    content.lastIndexOf("```")
                ).trim();
            } else if (content.contains("```")) {
                jsonContent = content.substring(
                    content.indexOf("```") + 3,
                    content.lastIndexOf("```")
                ).trim();
            }

            return objectMapper.readValue(jsonContent, responseType);
        } catch (Exception e) {
            log.error("解析 AI 响应失败: content={}", content, e);
            throw new RuntimeException("解析 AI 响应失败", e);
        }
    }

    /**
     * 简化的文本转接口（用于快速调用）
     */
    public Map<String, Object> chatToMap(String userMessage) {
        String content = chat(userMessage);

        try {
            return objectMapper.readValue(content, new com.fasterxml.jackson.core.type.TypeReference<Map<String, Object>>() {});
        } catch (Exception e) {
            log.error("解析 AI 响应失败: content={}", content, e);
            // 返回一个默认的 Map
            Map<String, Object> result = new HashMap<>();
            result.put("rawContent", content);
            return result;
        }
    }
}
```

package com.auction.service.ai;

import com.auction.api.dto.response.SuggestionResponse;
import com.auction.domain.entity.Auction;
import com.auction.domain.entity.Bid;
import com.auction.domain.entity.User;
import com.auction.repository.AuctionRepository;
import com.auction.repository.BidRepository;
import com.auction.repository.UserRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import java.math.BigDecimal;
import java.time.Duration;
import java.time.LocalDateTime;
import java.util.List;
import java.util.Map;

@Slf4j
@Service
@RequiredArgsConstructor
public class SuggestionService {

    private final DoubaoClient doubaoClient;
    private final AuctionRepository auctionRepository;
    private final BidRepository bidRepository;
    private final UserRepository userRepository;

    /**
     * 获取出价建议
     */
    public SuggestionResponse getSuggestion(Long auctionId, Long userId, BigDecimal maxBudget) {
        // 1. 获取竞拍信息
        Auction auction = auctionRepository.findById(auctionId);
        if (auction == null) {
            throw new com.auction.common.BizException(
                com.auction.common.ErrorCode.AUCTION_NOT_FOUND);
        }

        // 2. 获取用户信息
        User user = userRepository.findById(userId);
        String username = user != null ? user.getNickname() : "用户" + userId;

        // 3. 收集上下文数据
        BidContext context = collectBidContext(auction, userId, maxBudget);

        // 4. 构建 Prompt
        String prompt = buildPrompt(auction, context, username, maxBudget);

        try {
            // 5. 调用豆包 API
            Map<String, Object> result = doubaoClient.chatToMap(prompt);

            // 6. 解析响应
            return parseSuggestion(result, auction, context);

        } catch (Exception e) {
            log.error("获取出价建议失败: auctionId={}, userId={}", auctionId, userId, e);
            // 降级：返回基于规则的简单建议
            return getFallbackSuggestion(auction, context, maxBudget);
        }
    }

    /**
     * 收集出价上下文
     */
    private BidContext collectBidContext(Auction auction, Long userId, BigDecimal maxBudget) {
        // 当前价格
        BigDecimal currentPrice = auction.getCurrentPrice();

        // 出价次数
        List<Bid> recentBids = bidRepository.findRecentByAuctionId(auction.getId(), 100);
        int bidCount = recentBids.size();

        // 竞争人数
        long competitorCount = recentBids.stream()
            .map(Bid::getUserId)
            .distinct()
            .count();

        // 剩余时间
        long remainingSeconds = Duration.between(LocalDateTime.now(), auction.getEndTime()).getSeconds();

        // 用户上次出价
        BigDecimal userLastBid = null;
        List<Bid> userBids = bidRepository.findByAuctionIdAndUserId(auction.getId(), userId);
        if (!userBids.isEmpty()) {
            userLastBid = userBids.get(0).getAmount();
        }

        // 最近出价间隔
        double avgInterval = calculateAvgInterval(recentBids);

        return new BidContext(currentPrice, bidCount, competitorCount, remainingSeconds,
            userLastBid, avgInterval, recentBids);
    }

    /**
     * 计算平均出价间隔
     */
    private double calculateAvgInterval(List<Bid> bids) {
        if (bids.size() < 2) {
            return 0;
        }

        long totalInterval = 0;
        int count = 0;

        for (int i = 0; i < bids.size() - 1; i++) {
            long interval = Duration.between(bids.get(i + 1).getCreatedAt(),
                bids.get(i).getCreatedAt()).getSeconds();
            if (interval > 0 && interval < 3600) {  // 只统计1小时内的间隔
                totalInterval += interval;
                count++;
            }
        }

        return count > 0 ? (double) totalInterval / count : 0;
    }

    /**
     * 构建 Prompt
     */
    private String buildPrompt(Auction auction, BidContext context, String username, BigDecimal maxBudget) {
        String productName = auction.getProductName() != null ? auction.getProductName() : "商品";
        BigDecimal nextValidPrice = context.currentPrice.add(auction.getBidIncrement());

        return String.format("""

            - 当前价格：¥%.2f

            - 总出价次数：%d

            - 竞争人数：%d 人
            - 平均出价间隔：%.1f 秒/次

            ## 用户信息
            - 用户名：%s
            - 上次出价：%s
            - 预算上限：%s

            3. urgency 取值：NOW（立即出价）、WAIT（等待时机）、LAST_SEC（最后时刻）
            """,
            productName,

            nextValidPrice,

            context.bidCount,

            username,
            context.userLastBid != null ? "¥" + context.userLastBid : "无",
            maxBudget != null ? "¥" + maxBudget : "无限制",
            nextValidPrice
        );
    }

    /**
     * 解析 AI 响应
     */
    private SuggestionResponse parseSuggestion(Map<String, Object> result, Auction auction, BidContext context) {
        // 尝试从 Map 中提取数据
        BigDecimal suggestedPrice = extractPrice(result.get("suggestedPrice"));
        if (suggestedPrice == null) {
            suggestedPrice = context.currentPrice.add(auction.getBidIncrement());
        }

        String reason = extractString(result.get("reason"));
        if (reason == null || reason.isEmpty()) {
            reason = "基于当前竞拍情况的智能建议";
        }

        String urgency = extractString(result.get("urgency"));
        if (urgency == null || (!urgency.equals("NOW") && !urgency.equals("WAIT") && !urgency.equals("LAST_SEC"))) {
            urgency = context.remainingSeconds < 60 ? "LAST_SEC" : "WAIT";
        }

        Double winProbability = extractDouble(result.get("winProbability"));
        if (winProbability == null || winProbability < 0 || winProbability > 1) {
            winProbability = calculateWinProbability(context);
        }

        return SuggestionResponse.builder()
            .suggestedPrice(suggestedPrice)
            .reason(reason)
            .urgency(urgency)
            .winProbability(winProbability)
            .tips(extractTips(result.get("tips")))
            .build();
    }

    /**
     * 降级建议（基于规则）
     */
    private SuggestionResponse getFallbackSuggestion(Auction auction, BidContext context, BigDecimal maxBudget) {
        BigDecimal nextPrice = context.currentPrice.add(auction.getBidIncrement());

        String urgency;
        if (context.remainingSeconds < 30) {
            urgency = "LAST_SEC";
        } else if (context.competitorCount > 5 && context.avgInterval < 10) {
            urgency = "NOW";
        } else {
            urgency = "WAIT";
        }

        double winProb = calculateWinProbability(context);

        return SuggestionResponse.builder()
            .suggestedPrice(nextPrice)
            .reason("当前竞价激烈，建议按最低幅度出价")
            .urgency(urgency)
            .winProbability(winProb)
            .tips(List.of(
                "注意控制预算",
                "观察对手出价节奏",
                "剩余时间较少时可适当提高出价"
            ))
            .build();
    }

    /**
     * 计算胜率
     */
    private double calculateWinProbability(BidContext context) {
        // 简单的胜率计算逻辑
        double baseScore = 0.5;

        // 出价次数越少，胜率越高
        if (context.bidCount < 10) baseScore += 0.2;
        else if (context.bidCount < 30) baseScore += 0.1;

        // 竞争人数越少，胜率越高
        if (context.competitorCount < 5) baseScore += 0.2;
        else if (context.competitorCount < 10) baseScore += 0.1;

        // 剩余时间越少，胜率越低
        if (context.remainingSeconds < 30) baseScore -= 0.1;

        return Math.min(0.95, Math.max(0.05, baseScore));
    }

    private BigDecimal extractPrice(Object value) {
        if (value == null) return null;
        if (value instanceof BigDecimal) return (BigDecimal) value;
        if (value instanceof Number) return new BigDecimal(value.toString());
        try {
            return new BigDecimal(value.toString());
        } catch (Exception e) {
            return null;
        }
    }

    private String extractString(Object value) {
        if (value == null) return null;
        return value.toString();
    }

    private Double extractDouble(Object value) {
        if (value == null) return null;
        if (value instanceof Double) return (Double) value;
        if (value instanceof Number) return ((Number) value).doubleValue();
        try {
            return Double.parseDouble(value.toString());
        } catch (Exception e) {
            return null;
        }
    }

    @SuppressWarnings("unchecked")
    private java.util.List<String> extractTips(Object value) {
        if (value instanceof java.util.List) {
            return (java.util.List<String>) value;
        }
        return List.of();
    }

    /**
     * 出价上下文
     */
    private record BidContext(
        BigDecimal currentPrice,
        int bidCount,
        long competitorCount,
        long remainingSeconds,
        BigDecimal userLastBid,
        double avgInterval,
        List<Bid> recentBids
    ) {}
}

package com.auction.service.ai;

import com.auction.common.BizException;
import com.auction.common.ErrorCode;
import com.auction.domain.entity.Auction;
import com.auction.domain.entity.AutoBidConfig;
import com.auction.domain.enums.AuctionStatus;
import com.auction.repository.AuctionRepository;
import com.auction.repository.AutoBidConfigRepository;
import com.auction.service.bid.BidService;
import com.auction.service.websocket.WsMessageService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Service;
import java.math.BigDecimal;
import java.time.Duration;
import java.time.LocalDateTime;
import java.util.List;

@Slf4j
@Service
@RequiredArgsConstructor
public class AutoBidAgent {

    private final AutoBidConfigRepository configRepository;
    private final AuctionRepository auctionRepository;
    private final BidService bidService;
    private final WsMessageService wsMessageService;

    /**
     * 定时扫描代理出价任务
     * 每秒执行一次
     */
    @Scheduled(fixedRate = 1000)
    public void processAutoBids() {
        try {
            List<AutoBidConfig> activeConfigs = configRepository.findActiveAll();

            for (AutoBidConfig config : activeConfigs) {
                try {
                    processAutoBid(config);
                } catch (Exception e) {
                    log.error("处理代理出价失败: configId={}", config.getId(), e);
                }
            }
        } catch (Exception e) {
            log.error("扫描代理出价任务失败", e);
        }
    }

    /**
     * 处理单个代理出价
     */
    private void processAutoBid(AutoBidConfig config) {
        // 1. 获取竞拍信息
        Auction auction = auctionRepository.findById(config.getAuctionId());
        if (auction == null) {
            markCompleted(config, "竞拍不存在");
            return;
        }

        // 2. 检查竞拍状态
        if (auction.getStatusEnum() != AuctionStatus.ACTIVE) {
            if (auction.getStatusEnum() == AuctionStatus.COMPLETED ||
                auction.getStatusEnum() == AuctionStatus.CANCELLED) {
                markCompleted(config, "竞拍已结束");
            }
            return;
        }

        // 3. 检查是否超时
        if (LocalDateTime.now().isAfter(auction.getEndTime())) {
            markCompleted(config, "竞拍时间已过");
            return;
        }

        // 4. 根据策略决定是否出价
        boolean shouldBid = shouldBid(auction, config);

        if (shouldBid) {
            BigDecimal bidAmount = calculateBidAmount(auction, config);

            if (bidAmount.compareTo(config.getMaxPrice()) <= 0) {
                // 执行出价
                executeBid(config, bidAmount);
            } else {
                // 超出预算，停止代理
                markCompleted(config, "达到预算上限");
                notifyBudgetExceeded(config);
            }
        }
    }

    /**
     * 判断是否应该出价
     */
    private boolean shouldBid(Auction auction, AutoBidConfig config) {
        return switch (config.getStrategy()) {
            case "LAST_SEC" -> shouldBidLastSec(auction, config);
            case "SMART" -> shouldBidSmart(auction, config);
            case "AGGRESSIVE" -> shouldBidAggressive(auction, config);
            default -> false;
        };
    }

    /**
     * 最后时刻策略
     */
    private boolean shouldBidLastSec(Auction auction, AutoBidConfig config) {
        long remainingSec = Duration.between(LocalDateTime.now(), auction.getEndTime()).getSeconds();

        // 在最后 5 秒内，且不是当前最高出价者
        if (remainingSec <= 5 && remainingSec > 0) {
            return !isUserLeading(auction, config.getUserId());
        }
        return false;
    }

    /**
     * 智能策略
     */
    private boolean shouldBidSmart(Auction auction, AutoBidConfig config) {
        long remainingSec = Duration.between(LocalDateTime.now(), auction.getEndTime()).getSeconds();

        // 获取最近出价次数（估算）
        long recentBidCount = getRecentBidCount(auction.getId());

        if (recentBidCount > 10 && remainingSec < 30) {
            // 竞争激烈，提前出价
            return !isUserLeading(auction, config.getUserId());
        } else if (recentBidCount <= 3 && remainingSec <= 5) {
            // 竞争不激烈，最后出价
            return !isUserLeading(auction, config.getUserId());
        }

        return false;
    }

    /**
     * 激进策略
     */
    private boolean shouldBidAggressive(Auction auction, AutoBidConfig config) {
        // 只要不领先就出价
        return !isUserLeading(auction, config.getUserId());
    }

    /**
     * 计算出价金额
     */
    private BigDecimal calculateBidAmount(Auction auction, AutoBidConfig config) {
        BigDecimal currentPrice = auction.getCurrentPrice();
        BigDecimal increment = auction.getBidIncrement();
        BigDecimal nextPrice = currentPrice.add(increment);

        // 确保不超过最高预算
        BigDecimal maxBid = config.getMaxPrice();
        return nextPrice.min(maxBid);
    }

    /**
     * 判断用户是否领先
     */
    private boolean isUserLeading(Auction auction, Long userId) {
        return userId.equals(auction.getHighestBidder());
    }

    /**
     * 获取最近出价次数（简化版）
     */
    private long getRecentBidCount(Long auctionId) {
        // 这里简化处理，实际可以从 Redis 获取
        return 0;
    }

    /**
     * 执行出价
     */
    private void executeBid(AutoBidConfig config, BigDecimal amount) {
        try {
            com.auction.api.dto.request.PlaceBidRequest request =
                new com.auction.api.dto.request.PlaceBidRequest();
            request.setAuctionId(config.getAuctionId());
            request.setUserId(config.getUserId());
            request.setAmount(amount);
            request.setIsAutoBid(true);

            bidService.placeBid(request);

            // 更新配置
            config.setBidCount(config.getBidCount() + 1);
            config.setCurrentBid(amount);
            configRepository.save(config);

            log.info("自动出价成功: userId={}, auctionId={}, amount={}",
                config.getUserId(), config.getAuctionId(), amount);

        } catch (BizException e) {
            if (e.getErrorCode().equals(ErrorCode.BID_AMOUNT_TOO_LOW) ||
                e.getErrorCode().equals(ErrorCode.BID_AMOUNT_INVALID)) {
                // 价格已经更高了，下次继续尝试
                log.debug("自动出价被超越，下次继续: userId={}, auctionId={}",
                    config.getUserId(), config.getAuctionId());
            } else {
                log.error("自动出价失败: userId={}, auctionId={}, error={}",
                    config.getUserId(), config.getAuctionId(), e.getMessage());
                // 其他错误可能需要停止代理
                if (e.getErrorCode().equals(ErrorCode.AUCTION_ALREADY_ENDED) ||
                    e.getErrorCode().equals(ErrorCode.AUCTION_CANCELLED)) {
                    markCompleted(config, "竞拍状态变更");
                }
            }
        }
    }

    /**
     * 标记完成
     */
    private void markCompleted(AutoBidConfig config, String reason) {
        config.setStatus("COMPLETED");
        configRepository.save(config);
        log.info("代理出价完成: userId={}, auctionId={}, reason={}",
            config.getUserId(), config.getAuctionId(), reason);
    }

    /**
     * 通知用户预算已用尽
     */
    private void notifyBudgetExceeded(AutoBidConfig config) {
        wsMessageService.sendYouLost(config.getUserId(), config.getAuctionId());
    }

    // ==================== 公开方法 ====================

    /**
     * 设置代理出价
     */
    public AutoBidConfig setAutoBid(Long userId, Long auctionId, BigDecimal maxPrice, String strategy) {
        // 检查是否已存在
        AutoBidConfig existing = configRepository.findByUserAndAuction(userId, auctionId);
        if (existing != null) {
            // 更新
            existing.setMaxPrice(maxPrice);
            existing.setStrategy(strategy);
            existing.setStatus("ACTIVE");
            existing.setBidCount(0);
            existing.setCurrentBid(null);
            return configRepository.save(existing);
        }

        // 创建新的
        AutoBidConfig config = new AutoBidConfig();
        config.setUserId(userId);
        config.setAuctionId(auctionId);
        config.setMaxPrice(maxPrice);
        config.setStrategy(strategy);
        config.setStatus("ACTIVE");
        config.setBidCount(0);

        return configRepository.save(config);
    }

    /**
     * 取消代理出价
     */
    public void cancelAutoBid(Long userId, Long auctionId) {
        AutoBidConfig config = configRepository.findByUserAndAuction(userId, auctionId);
        if (config != null) {
            config.setStatus("CANCELLED");
            configRepository.save(config);
        }
    }

    /**
     * 获取代理出价状态
     */
    public AutoBidConfig getAutoBidStatus(Long userId, Long auctionId) {
        return configRepository.findByUserAndAuction(userId, auctionId);
    }
}

        this.description = description;

        return LOW;

package com.auction.service.risk;

import com.auction.domain.entity.Bid;
import com.auction.domain.entity.UserBehavior;

    private final BidRepository bidRepository;
    private final RedisService redisService;

     * 记录出价行为
     */
    public void recordBid(Long userId, Long auctionId, BigDecimal amount) {
        UserBehavior behavior = behaviorRepository.findByUserAndAuction(userId, auctionId);

        if (behavior == null) {
            behavior = new UserBehavior();
            behavior.setUserId(userId);
            behavior.setAuctionId(auctionId);
            behavior.setBidCount(1);
            behavior.setLastBidTime(LocalDateTime.now());
            behavior.setRiskScore(BigDecimal.ZERO);
            behavior.setRiskLevel("LOW");
            behavior.setIsBlocked(false);
        } else {
            // 更新平均间隔

            behavior.setBidCount(userBids.size());
            behavior.setLastBidTime(LocalDateTime.now());

            if (userBids.size() >= 2) {
                long totalInterval = 0;
                int count = 0;
                for (int i = 0; i < userBids.size() - 1; i++) {

                        userBids.get(i + 1).getCreatedAt(),
                        userBids.get(i).getCreatedAt()
                    ).getSeconds();
                    if (interval > 0 && interval < 3600) {
                        totalInterval += interval;
                        count++;
                    }

                behavior.setAvgBidInterval(count > 0 ? (int) (totalInterval / count) : 0);
            }
        }

        behaviorRepository.save(behavior);

```java
package com.auction.service.risk;

import com.auction.api.dto.response.RiskCheckResult;
import com.auction.common.BizException;
import com.auction.common.ErrorCode;
import com.auction.domain.entity.Auction;
import com.auction.domain.entity.User;
import com.auction.domain.entity.UserBehavior;
import com.auction.domain.enums.RiskAction;
import com.auction.domain.enums.RiskLevel;
import com.auction.repository.AuctionRepository;
import com.auction.repository.RiskEventRepository;
import com.auction.repository.UserRepository;
import com.auction.infrastructure.redis.RedisService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.time.temporal.ChronoUnit;
import java.util.ArrayList;
import java.util.List;

@Slf4j
@Service
@RequiredArgsConstructor
public class RiskDetectionService {

    private final UserBehaviorService userBehaviorService;
    private final UserRepository userRepository;
    private final AuctionRepository auctionRepository;
    private final RiskEventRepository riskEventRepository;
    private final RedisService redisService;

    /**
     * 检查出价风险
     */
    public RiskCheckResult checkBidRisk(Long userId, Long auctionId, BigDecimal amount) {
        List<String> reasons = new ArrayList<>();
        double totalScore = 0;
        double totalWeight = 0;

        // 1. 检查用户是否被封禁
        if (userBehaviorService.isBlocked(userId, auctionId)) {
            RiskCheckResult result = RiskCheckResult.builder()
                .score(1.0)
                .level(RiskLevel.CRITICAL)
                .action(RiskAction.BLOCK)
                .reasons(List.of("账户已被限制"))
                .blocked(true)
                .build();
            recordRiskEvent(userId, auctionId, "USER_BLOCKED", RiskLevel.CRITICAL, "账户被封禁", RiskAction.BLOCK);
            return result;
        }

        // 2. 频率检测
        double[] freqResult = checkFrequency(userId, auctionId);
        totalScore += freqResult[0] * freqResult[1];
        totalWeight += freqResult[1];
        if (freqResult[0] > 0.5) {
            reasons.add("出价频率异常");
        }

        // 3. 金额模式检测
        double[] amountResult = checkAmountPattern(userId, auctionId);
        totalScore += amountResult[0] * amountResult[1];
        totalWeight += amountResult[1];
        if (amountResult[0] > 0.5) {
            reasons.add("出价金额异常");
        }

        // 4. 新用户检测
        double[] newUserResult = checkNewUser(userId);
        totalScore += newUserResult[0] * newUserResult[1];
        totalWeight += newUserResult[1];
        if (newUserResult[0] > 0.5) {
            reasons.add("新用户高风险行为");
        }

        // 5. 支付历史检测
        double[] paymentResult = checkPaymentHistory(userId);
        totalScore += paymentResult[0] * paymentResult[1];
        totalWeight += paymentResult[1];
        if (paymentResult[0] > 0.5) {
            reasons.add("历史弃单率高");
        }

        // 计算最终分数
        double finalScore = totalWeight > 0 ? totalScore / totalWeight : 0;
        RiskLevel level = RiskLevel.fromScore(finalScore);
        RiskAction action = RiskAction.fromLevel(level);

        // 记录风控事件
        if (level != RiskLevel.LOW) {
            recordRiskEvent(userId, auctionId, "BID_RISK", level,
                String.join("; ", reasons), action);
        }

        // 更新用户风险评分
        userBehaviorService.updateRiskScore(userId, auctionId, finalScore, level.name());

        RiskCheckResult result = RiskCheckResult.builder()
            .score(finalScore)
            .level(level)
            .action(action)
            .reasons(reasons)
            .blocked(action == RiskAction.BLOCK || action == RiskAction.MANUAL_REVIEW)
            .build();

        log.info("风控检测完成: userId={}, auctionId={}, score={}, level={}, action={}",
            userId, auctionId, finalScore, level, action);

        return result;
    }

    /**
     * 频率检测
     */
    private double[] checkFrequency(Long userId, Long auctionId) {
        String key = "bid:count:" + userId + ":1min";
        Object countObj = redisService.get(key);
        int count = countObj != null ? Integer.parseInt(countObj.toString()) : 0;

        double weight = 0.4;
        double score;

        if (count > 20) {
            score = 0.9;  // 极高风险
        } else if (count > 10) {
            score = 0.6;  // 中等风险
        } else if (count > 5) {
            score = 0.3;  // 轻微风险
        } else {
            score = 0.0;  // 正常
        }

        return new double[]{score, weight};
    }

    /**
     * 金额模式检测
     */
    private double[] checkAmountPattern(Long userId, Long auctionId) {
        UserBehavior behavior = userBehaviorService.getUserBehavior(userId, auctionId);
        double weight = 0.2;
        double score = 0.0;

        if (behavior != null && behavior.getAvgBidInterval() != null) {
            // 平均间隔小于3秒，疑似机器人
            if (behavior.getAvgBidInterval() < 3) {
                score = 0.7;
            }
        }

        return new double[]{score, weight};
    }

    /**
     * 新用户检测
     */
    private double[] checkNewUser(Long userId) {
        User user = userRepository.findById(userId);
        double weight = 0.2;
        double score = 0.0;

        if (user != null && user.getCreatedAt() != null) {
            long hoursSinceRegister = ChronoUnit.HOURS.between(
                user.getCreatedAt(), LocalDateTime.now());

            if (hoursSinceRegister < 1) {
                // 注册1小时内 + 高频出价 = 高风险
                // 这里简化处理
                score = 0.4;
            } else if (hoursSinceRegister < 24) {
                score = 0.2;
            }
        }

        return new double[]{score, weight};
    }

    /**
     * 支付历史检测
     */
    private double[] checkPaymentHistory(Long userId) {
        // 简化版：假设新用户风险较低
        // 实际应该查询订单表计算弃单率
        double weight = 0.2;
        double score = 0.0;
        return new double[]{score, weight};
    }

    /**
     * 记录风控事件
     */
    private void recordRiskEvent(Long userId, Long auctionId, String eventType,
                                   RiskLevel severity, String description, RiskAction action) {
        try {
            com.auction.domain.entity.RiskEvent event = new com.auction.domain.entity.RiskEvent();
            event.setUserId(userId);
            event.setAuctionId(auctionId);
            event.setEventType(eventType);
            event.setSeverity(severity.name());
            event.setDescription(description);
            event.setActionTaken(action.name());
            event.setHandledBy("SYSTEM");

            riskEventRepository.save(event);
        } catch (Exception e) {
            log.error("记录风控事件失败", e);
        }
    }

    /**
     * 封禁用户
     */
    public void blockUser(Long userId, Long auctionId, String reason) {
        userBehaviorService.setBlocked(userId, auctionId, true);
        recordRiskEvent(userId, auctionId, "MANUAL_BLOCK", RiskLevel.HIGH, reason, RiskAction.BLOCK);
    }

    /**
     * 解封用户
     */
    public void unblockUser(Long userId, Long auctionId) {
        userBehaviorService.setBlocked(userId, auctionId, false);
    }
}
