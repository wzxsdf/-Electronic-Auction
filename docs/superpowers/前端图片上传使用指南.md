# 前端图片上传使用指南

## 🎯 两种上传方式对比

### 方式1：前端直接上传OSS（推荐）

**优势**：
- ✅ 不占用服务器带宽
- ✅ 减轻服务器压力
- ✅ 上传速度更快
- ✅ 适合大文件

**流程**：
```
1. 前端请求 /upload/token 获取STS凭证
2. 前端使用阿里云OSS SDK直接上传
3. 上传成功后获取图片URL
4. 提交商品信息时包含图片URL
```

### 方式2：后端代理上传（备用）

**优势**：
- ✅ 便于控制和管理
- ✅ 可以进行额外的处理
- ✅ 适合小文件

**流程**：
```
1. 前端直接上传文件到 /upload/image
2. 后端验证并转发到OSS
3. 返回图片URL给前端
```

import OSS from 'ali-oss';

/**
 * 方式1：前端直接上传到OSS（推荐）
 */
async function uploadImageToOSS(file) {
  try {
    // 1. 获取上传凭证
    const tokenResponse = await fetch('/upload/token?fileName=' + file.name, {
      method: 'POST',
      headers: {
        'Authorization': 'Bearer ' + getToken(),
        'Content-Type': 'application/json'
      }
    });

    if (!tokenResponse.ok) {
      throw new Error('获取上传凭证失败');
    }

    const tokenData = await tokenResponse.json();
    const token = tokenData.data;

    // 2. 初始化OSS客户端
    const client = new OSS({
      region: token.region,
      accessKeyId: token.accessKeyId,
      accessKeySecret: token.accessKeySecret,
      stsToken: token.stsToken,
      bucket: token.bucket,
      secure: true // 使用HTTPS
    });

    // 3. 生成唯一文件名
    const fileName = `${Date.now()}_${Math.random().toString(36).substring(7)}_${file.name}`;
    const objectKey = token.keyPrefix + fileName;

    // 4. 上传文件
    const result = await client.put(objectKey, file);

    // 5. 返回图片URL
    console.log('上传成功:', result.url);
    return result.url;

  } catch (error) {
    console.error('上传失败:', error);
    throw error;
  }
}

/**
 * 方式2：后端代理上传（备用）
 */
async function uploadImageViaBackend(file) {
  try {
    const formData = new FormData();
    formData.append('file', file);

    const response = await fetch('/upload/image', {
      method: 'POST',
      headers: {
        'Authorization': 'Bearer ' + getToken()
      },
      body: formData
    });

    if (!response.ok) {
      throw new Error('上传失败');
    }

    const data = await response.json();
    console.log('上传成功:', data.data.imageUrl);
    return data.data.imageUrl;

  } catch (error) {
    console.error('上传失败:', error);
    throw error;
  }
}

### 2. Vue.js 示例

```vue
<template>
  <div class="image-upload">
    <!-- 文件选择 -->
    <el-upload
      :auto-upload="false"
      :on-change="handleFileChange"
      :show-file-list="false"
      accept="image/*"
    >
      <el-button type="primary">选择图片</el-button>
    </el-upload>

    <!-- 上传进度 -->
    <el-progress v-if="uploading" :percentage="uploadProgress" />

    <!-- 图片预览 -->
    <img v-if="imageUrl" :src="imageUrl" style="max-width: 200px;" />

    <!-- 提交按钮 -->
    <el-button @click="submitProduct" type="success">创建商品</el-button>
  </div>
</template>

<script>
import OSS from 'ali-oss';

export default {
  data() {
    return {
      uploading: false,
      uploadProgress: 0,
      imageUrl: ''
    };
  },

  methods: {
    async handleFileChange(file) {
      // 验证文件
      if (!file.raw.type.startsWith('image/')) {
        this.$message.error('只能上传图片文件');
        return;
      }

      if (file.raw.size > 10 * 1024 * 1024) {
        this.$message.error('文件大小不能超过10MB');
        return;
      }

      this.uploading = true;
      this.uploadProgress = 0;

      try {
        // 上传图片
        this.imageUrl = await this.uploadImageToOSS(file.raw);
        this.$message.success('上传成功');

      } catch (error) {
        this.$message.error('上传失败: ' + error.message);
      } finally {
        this.uploading = false;
        this.uploadProgress = 0;
      }
    },

    async uploadImageToOSS(file) {
      // 1. 获取上传凭证
      const { data: token } = await this.$http.post('/upload/token', null, {
        params: { fileName: file.name }
      });

      // 2. 初始化OSS客户端
      const client = new OSS({
        region: token.region,
        accessKeyId: token.accessKeyId,
        accessKeySecret: token.accessKeySecret,
        stsToken: token.stsToken,
        bucket: token.bucket,
        secure: true
      });

      // 3. 上传文件
      const fileName = `${Date.now()}_${file.name}`;
      const objectKey = token.keyPrefix + fileName;

      const result = await client.put(objectKey, file);

      return result.url;
    },

    async submitProduct() {
      if (!this.imageUrl) {
        this.$message.warning('请先上传图片');
        return;
      }

      const productData = {
        merchantId: 1,
        name: '商品名称',
        imageUrl: this.imageUrl,  // 使用上传后的URL
        description: '商品描述',
        category: '手机数码',
        initialPrice: 7999.00
      };

      await this.$http.post('/products', productData);
      this.$message.success('商品创建成功');
    }
  }
};
</script>
```

async function uploadMultipleImages(files) {
  const formData = new FormData();

  // 添加所有文件

      method: 'POST',
      headers: {
        'Authorization': 'Bearer ' + getToken()

    const data = await response.json();

    console.log('上传成功:', data.data.successUrls);

    return data.data;

  } catch (error) {

import * as ImagePicker from 'expo-image-picker';
import OSS from 'ali-oss';

/**
 * 选择并上传图片
 */
async function pickAndUploadImage() {
  // 1. 选择图片
  const result = await ImagePicker.launchImageLibraryAsync({
    mediaTypes: ImagePicker.MediaTypeOptions.Images,
    allowsEditing: true,
    aspect: [16, 9],
    quality: 1,
  });

  if (!result.canceled) {
    const asset = result.assets[0];

    // 2. 上传图片
    try {
      const imageUrl = await uploadImageToOSS(asset.uri);
      console.log('上传成功:', imageUrl);
      return imageUrl;

    } catch (error) {
      console.error('上传失败:', error);
      throw error;
    }
  }
}

/**
 * 上传到OSS
 */
async function uploadImageToOSS(fileUri) {
  // 1. 获取上传凭证
  const tokenResponse = await fetch('/upload/token');
  const token = (await tokenResponse.json()).data;

  // 2. 读取文件
  const fileInfo = await FileSystem.getInfoAsync(fileUri);
  const fileContent = await FileSystem.readAsStringAsync(fileUri, {
    encoding: FileSystem.EncodingType.Base64,
  });

  // 3. 上传到OSS
  const client = new OSS({
    region: token.region,
    accessKeyId: token.accessKeyId,
    accessKeySecret: token.accessKeySecret,
    stsToken: token.stsToken,
    bucket: token.bucket,
  });

  const fileName = fileUri.split('/').pop();
  const objectKey = token.keyPrefix + Date.now() + '_' + fileName;

  const result = await client.put(objectKey, fileContent, {
    headers: {
      'Content-Type': 'image/jpeg'
    }
  });

  return result.url;
}

## 🎨 上传组件示例（带进度条）

```javascript
import React, { useState } from 'react';
import { Upload, Button, Progress, message } from 'antd';
import { UploadOutlined } from '@ant-design/icons';

function ImageUploadWithProgress() {
  const [fileList, setFileList] = useState([]);
  const [uploading, setUploading] = useState(false);
  const [progress, setProgress] = useState(0);

  const handleUpload = async (file) => {
    setUploading(true);
    setProgress(0);

    try {
      // 模拟上传进度
      const interval = setInterval(() => {
        setProgress(prev => {
          if (prev >= 90) {
            clearInterval(interval);
            return 90;
          }
          return prev + 10;
        });
      }, 100);

      // 上传图片（这里使用后端代理上传方式）
      const formData = new FormData();
      formData.append('file', file);

      const response = await fetch('/upload/image', {
        method: 'POST',
        headers: {
          'Authorization': 'Bearer ' + getToken()
        },
        body: formData
      });

      clearInterval(interval);
      setProgress(100);

      const data = await response.json();

      if (data.code === 200) {
        setFileList([...fileList, {
          uid: file.uid,
          name: file.name,
          status: 'done',
          url: data.data.imageUrl,
        }]);

        message.success('上传成功');
        return data.data.imageUrl;
      } else {
        throw new Error(data.message);
      }

    } catch (error) {
      message.error('上传失败: ' + error.message);
      throw error;
    } finally {
      setUploading(false);
      setProgress(0);
    }
  };

  const uploadProps = {
    beforeUpload: (file) => {
      // 验证文件类型
      const isImage = file.type.startsWith('image/');
      if (!isImage) {
        message.error('只能上传图片文件');
        return false;
      }

      // 验证文件大小
      const isLt10M = file.size / 1024 / 1024 < 10;
      if (!isLt10M) {
        message.error('图片大小不能超过10MB');
        return false;
      }

      // 自定义上传
      handleUpload(file);
      return false; // 阻止默认上传行为
    },

    onRemove: (file) => {
      setFileList(fileList.filter(item => item.uid !== file.uid));
    },

    fileList: fileList,
  };

  return (
    <div>
      <Upload {...uploadProps} listType="picture-card">
        {fileList.length >= 8 ? null : (
          <div>
            <UploadOutlined />
            <div style={{ marginTop: 8 }}>上传图片</div>
          </div>
        )}
      </Upload>

      {uploading && (
        <Progress
          percent={progress}
          status="active"
          style={{ marginTop: 16 }}
        />
      )}
    </div>
  );
}
```

## 🚀 性能优化建议

### 1. 图片压缩
```javascript
/**
 * 压缩图片后再上传
 */
async function compressAndUpload(file) {
  const canvas = document.createElement('canvas');
  const ctx = canvas.getContext('2d');
  const img = new Image();

  img.onload = async () => {
    // 设置压缩后的尺寸
    const maxWidth = 1920;
    const maxHeight = 1080;
    let width = img.width;
    let height = img.height;

    if (width > height) {
      if (width > maxWidth) {
        height *= maxWidth / width;
        width = maxWidth;
      }
    } else {
      if (height > maxHeight) {
        width *= maxHeight / height;
        height = maxHeight;
      }
    }

    canvas.width = width;
    canvas.height = height;
    ctx.drawImage(img, 0, 0, width, height);

    // 压缩并上传
    canvas.toBlob(async (blob) => {
      const compressedFile = new File([blob], file.name, {
        type: 'image/jpeg',
        lastModified: Date.now()
      });

      await uploadImageToOSS(compressedFile);
    }, 'image/jpeg', 0.8); // 80%质量
  };

  img.src = URL.createObjectURL(file);
}
```

/**
 * 使用OSS分片上传（适合大文件）
 */
async function multipartUpload(file) {
  const token = await getUploadToken();

  const client = new OSS({
    region: token.region,
    accessKeyId: token.accessKeyId,
    accessKeySecret: token.accessKeySecret,
    stsToken: token.stsToken,
    bucket: token.bucket,
  });

  try {
    // 分片上传
    const result = await client.multipartUpload(
      token.keyPrefix + file.name,
      file,
      {
        parallel: 3,  // 并发分片数
        partSize: 1024 * 1024,  // 每片1MB
        progress: (p, checkpoint) => {
          console.log('上传进度:', p);
        }
      }
    );

    console.log('上传成功:', result.url);
    return result.url;

  } catch (error) {
    console.error('上传失败:', error);
    throw error;
  }
}
