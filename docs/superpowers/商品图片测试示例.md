## JavaScript 测试代码

### 使用 Fetch API
```javascript
// 创建商品
async function createProduct() {
  const response = await fetch('http://localhost:8080/products', {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer ' + yourJwtToken,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      name: '测试商品',
      imageUrls: [
        'https://example.com/image1.jpg',
        'https://example.com/image2.jpg',
        'https://example.com/image3.jpg'
      ],
      initialPrice: 100.00
    })
  });
  
  const data = await response.json();
  console.log('创建结果:', data);
  return data;
}

// 查询商品
async function getProduct(productId) {
  const response = await fetch(`http://localhost:8080/products/${productId}`);
  const data = await response.json();
  
  if (data.code === 200) {
    console.log('商品图片列表:', data.data.imageUrls);
    console.log('主图URL:', data.data.primaryImageUrl);
  }
  
  return data;
}
```

### 使用 Axios
```javascript
// 创建商品
const createProduct = async () => {
  try {
    const response = await axios.post('/products', {
      name: 'Apple iPhone 15 Pro Max',
      brand: 'Apple',
      imageUrls: [
        'https://example.com/image1.jpg',
        'https://example.com/image2.jpg',

      ],
      description: '全新iPhone 15 Pro Max',
      initialPrice: 9999.00
    }, {
      headers: {
        'Authorization': `Bearer ${yourJwtToken}`
      }
    });
    
    console.log('创建成功:', response.data);
    return response.data;
  } catch (error) {
    console.error('创建失败:', error.response.data);
  }
};

// 查询商品并显示图片
const showProductImages = async (productId) => {
  const response = await axios.get(`/products/${productId}`);
  const product = response.data.data;
  
  console.log('主图:', product.primaryImageUrl);
  console.log('所有图片:', product.imageUrls);
  
  // 在页面上显示图片
  product.imageUrls.forEach(url => {
    const img = document.createElement('img');
    img.src = url;
    img.alt = product.name;
    document.body.appendChild(img);
  });
};
```

## React 组件示例

```jsx
import React, { useState } from 'react';
import axios from 'axios';

function ProductForm() {
  const [formData, setFormData] = useState({
    name: '',
    imageUrls: [],
    description: '',
    initialPrice: ''
  });

  const [imageInput, setImageInput] = useState('');

  // 添加图片URL
  const addImageUrl = () => {
    if (imageInput && formData.imageUrls.length < 4) {
      setFormData({
        ...formData,
        imageUrls: [...formData.imageUrls, imageInput]
      });
      setImageInput('');
    }
  };

  // 删除图片URL
  const removeImageUrl = (index) => {
    setFormData({
      ...formData,
      imageUrls: formData.imageUrls.filter((_, i) => i !== index)
    });
  };

  // 提交表单
  const handleSubmit = async (e) => {
    e.preventDefault();
    
    try {
      const response = await axios.post('/products', formData, {
        headers: {
          'Authorization': `Bearer ${yourJwtToken}`
        }
      });
      
      alert('商品创建成功！');
      console.log('响应数据:', response.data);
    } catch (error) {
      alert('创建失败：' + error.response.data.message);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>商品名称</label>
        <input
          type="text"
          value={formData.name}
          onChange={(e) => setFormData({ ...formData, name: e.target.value })}
          required
        />
      </div>

      <div>
        <label>图片URL（最多4张）</label>
        <div>
          <input
            type="url"
            value={imageInput}
            onChange={(e) => setImageInput(e.target.value)}
            placeholder="https://example.com/image.jpg"
          />
          <button type="button" onClick={addImageUrl}>
            添加 ({formData.imageUrls.length}/4)
          </button>
        </div>
        
        {/* 图片列表预览 */}
        <div>
          {formData.imageUrls.map((url, index) => (
            <div key={index}>
              <img src={url} alt={`预图 ${index + 1}`} style={{ width: '100px' }} />
              <button type="button" onClick={() => removeImageUrl(index)}>
                删除
              </button>
            </div>
          ))}
        </div>
      </div>

      <div>
        <label>描述</label>
        <textarea
          value={formData.description}
          onChange={(e) => setFormData({ ...formData, description: e.target.value })}
        />
      </div>

      <div>
        <label>价格</label>
        <input
          type="number"
          value={formData.initialPrice}
          onChange={(e) => setFormData({ ...formData, initialPrice: e.target.value })}
        />
      </div>

      <button type="submit">创建商品</button>
    </form>
  );
}

export default ProductForm;
```

## Vue 组件示例

```vue
<template>
  <div>
    <h2>创建商品</h2>
    <form @submit.prevent="handleSubmit">
      <div>
        <label>商品名称</label>
        <input v-model="form.name" required />
      </div>

      <div>
        <label>图片URL（最多4张）</label>
        <div>
          <input
            v-model="imageInput"
            type="url"
            placeholder="https://example.com/image.jpg"
          />
          <button type="button" @click="addImageUrl">
            添加 ({{ form.imageUrls.length }}/4)
          </button>
        </div>
        
        <!-- 图片预览 -->
        <div v-for="(url, index) in form.imageUrls" :key="index">
          <img :src="url" :alt="'图片' + (index + 1)" style="width: 100px" />
          <button type="button" @click="removeImageUrl(index)">删除</button>
        </div>
      </div>

      <div>
        <label>价格</label>
        <input v-model="form.initialPrice" type="number" />
      </div>

      <button type="submit">创建商品</button>
    </form>
  </div>
</template>

<script>
import axios from 'axios';

export default {
  data() {
    return {
      form: {
        name: '',
        imageUrls: [],
        initialPrice: ''
      },
      imageInput: ''
    };
  },
  methods: {
    addImageUrl() {
      if (this.imageInput && this.form.imageUrls.length < 4) {
        this.form.imageUrls.push(this.imageInput);
        this.imageInput = '';
      }
    },
    removeImageUrl(index) {
      this.form.imageUrls.splice(index, 1);
    },
    async handleSubmit() {
      try {
        const response = await axios.post('/products', this.form, {
          headers: {
            'Authorization': `Bearer ${yourJwtToken}`
          }
        });
        alert('创建成功！');
        console.log(response.data);
      } catch (error) {
        alert('创建失败：' + error.response.data.message);
      }
    }
  }
};
</script>
```
